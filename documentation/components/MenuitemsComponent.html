export class MenuitemsComponent implements OnInit {

  // @HostBinding('@pageAnimations')

  p: any //html page
  items = [];
  pageOfItems: Array<any>;
  lengthOfArray: number
  currentPage: any; //paging component

  statusmessage  = '';
  section = 1;

  menuItems$:   Observable<IMenuItem[]>
  menuItems: IMenuItem[];

  itemsPerPage = 25;

  id:         string;
  bucketName: string;

  // @HostListener('document:mousemove', ['$event'])
  // onMouseMove(e) {
  //   console.log(e);
  // }
  globalListenFunc: Function;

  constructor(private menuService: MenuService,
      private awsBucketService: AWSBucketService,
      private router: Router,
      public route: ActivatedRoute,
      private siteService: SitesService,
      private renderer: Renderer2

      )  {

      this.id = this.route.snapshot.paramMap.get('id');

}

  async ngOnInit()  {
    this.bucketName =   await this.awsBucketService.awsBucket();
    this.setItemsPerPage()
    this.initList(this.id)
    this.initListener();
  }

  initListener() {
    this.globalListenFunc = this.renderer.listen('document', 'mousemove', e => {
      console.log(e);
    });
  }

  // @HostListener("window:resize", []) updateItemsPerPage() {

  //   this.itemsPerPage = 15; // lg

  //   this.setItemsPerPage()

  //   const currentPage = this.currentPage
  //   this.initList(this.id  )
  //   this.currentPage = currentPage
  // }

  setItemsPerPage() {

    this.itemsPerPage = 15

    if (window.innerWidth >= 1200) {
      this.itemsPerPage = 15; // lg

    } else if (window.innerWidth >= 992) {
      this.itemsPerPage = 12;//md

    } else if (window.innerWidth  >= 768) {
      this.itemsPerPage = 12;//sm

    } else if (window.innerWidth < 768) {
      this.itemsPerPage = 8;//xs

    } else if (window.innerWidth < 600) {
      this.itemsPerPage = 5;//xs

    }
  }

  initList(id: string) : any {

      if(id)
      {
        const productSearchModel = {} as ProductSearchModel;
        productSearchModel.categoryID  = id;
        productSearchModel.pageNumber  = 1
        productSearchModel.pageSize    = 150
        const site                     = this.siteService.getAssignedSite();
        this.menuItems$                = this.menuService.getProductsBySearch(site, productSearchModel);

      } else {
        this.router.navigate(["/categories/"]);
        this.statusmessage = "ID is missing. Please go to categories"
        return this.statusmessage
      }

  };


  listItem(id:number) {
    this.router.navigate(["/menuitem/", {id:id}]);
  }

  getItemSrc(item:IMenuItem) {
    if (!item.urlImageMain) {
      return this.awsBucketService.getImageURLPath(this.bucketName, "placeholderproduct.jpg")
    } else {
      const arrary = item.urlImageMain.split(",")
      return this.awsBucketService.getImageURLPath(this.bucketName,  arrary[0] )
    }
  }

}
