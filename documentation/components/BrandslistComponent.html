export class BrandslistComponent implements OnInit {
  // @HostBinding('@pageAnimations')

  isDown =                false;
  startX:                 any;
  scrollLeft:             any;
  href:                   string;
  brandslider:            HTMLElement;
  section:                number;
  bucketName:             string;
  bucket:                 string;
  brands$:                Observable<IUserProfile[]>;
  brands:                 IUserProfile[];
  singlePage:             boolean;
  platForm =  this.getPlatForm();

  constructor(
      private contactsService: ContactsService,
      private awsBucket: AWSBucketService,
      private menuService: MenuService,
      private router: Router,
      private siteService: SitesService,)  {

    this.section = 1;
    this.href = this.router.url;

    if (this.href === '/brandslist') {
      this.singlePage = true
    } {
      this.singlePage = false
    }
  }

 getPlatForm() {  return Capacitor.getPlatform(); }

 async ngOnInit()  {
    await this.getBucket()
    const site = this.siteService.getAssignedSite();
    this.brands$ = this.contactsService.getBrands(site)
    this.brands = await this.brands$.pipe().toPromise();
    this.brandslider = document.querySelector('.items')  as HTMLElement
  }

  async getBucket() {
    this.bucket =   await this.awsBucket.awsBucket();
    if (this.bucket == null) {
      this.bucket = 'naturesherbs'
    }
  }

  @HostListener('mousedown', ['$event'])  onMouseDown(event: any) {
    if (this.href === '/brandslist') { return}
    if (!this.brandslider) { this.brandslider = document.querySelector('.branditems')  as HTMLElement }
    if ( this.brandslider) {
      this.isDown = true;
      this.brandslider.classList.add('active');
      this.startX = event.pageX - this.brandslider.offsetLeft;
      this.scrollLeft = this.brandslider.scrollLeft;
    }

  }

  @HostListener('mouseleave', ['$event']) onMouseLeave(event: any) {
    if (this.href === '/brandslist') { return}
    if (!this.brandslider) {         this.brandslider = document.querySelector('.branditems')  as HTMLElement }
    if ( this.brandslider) {
      this.isDown = false;
      this.brandslider.classList.remove('active');
    }
  }

  @HostListener('mouseup', ['$event']) onMouseUp(event: any) {
    if (this.href === '/brandslist') { return}
    if (!this.brandslider) { this.brandslider = document.querySelector('.branditems')  as HTMLElement}
    if ( this.brandslider) {
    this.isDown = false;
    this.brandslider.classList.remove('active');
    }
  }

  @HostListener('mousemove', ['$event']) onMouseMove(event: any) {
    if (this.href == '/brandslist') { return}
    if (!this.brandslider) { this.brandslider = document.querySelector('.branditems')  as HTMLElement }
    if(!this.isDown) {return;}
    if (this.brandslider) {
      event.preventDefault();
      const x = event.pageX - this.brandslider.offsetLeft;
      const walk = (x -  this.startX) * 3; //scroll-fast
      this.brandslider.scrollLeft =  this.scrollLeft - walk;
    }
  }


  showAll() {
    this.router.navigate(["/brandslist/"]);
  }

  getBrandSource(item: IUserProfile) {
    return this.getItemSrc(item.onlineDescriptionImage)
  }

  getItemSrc(nameArray: string) {
    return this.awsBucket.getImageURLFromNameArray(this.bucket, nameArray)
  }

  /* lists items in BrandComponent */
  listItems(id:number) {
    this.router.navigate(["/menuitems-infinite/", {brandID:id}]);
  }

  initProductSearchModel(id: number): ProductSearchModel {
    let productSearchModel        = {} as ProductSearchModel;
    { productSearchModel.brandID  = id.toString(); }
    productSearchModel.pageSize   = 25
    productSearchModel.pageNumber = 1
    console.log(productSearchModel)
    this.menuService.updateMeunuItemData(productSearchModel)
    return productSearchModel
  }

}
