import { ChangeDetectionStrategy, Component, Input, OnChanges, OnInit, SimpleChanges } from '@angular/core';
import { AWSBucketService, ContactsService, MenuService } from 'src/app/_services';
import { IClientTable, IUserProfile }  from 'src/app/_interfaces';
import { ActivatedRoute, Router } from '@angular/router';
import { DevService } from 'src/app/_services/system/dev-service.service';
import { FormBuilder, FormGroup, FormControl, FormArray, Validator } from '@angular/forms';
import { Subject } from 'rxjs';
import { DomSanitizer } from '@angular/platform-browser';
import { ClientTableService } from 'src/app/_services/people/client-table.service';
import { MatSnackBar } from '@angular/material/snack-bar';
import { SitesService } from 'src/app/_services/reporting/sites.service';

@Component({
  selector: 'app-adminbranditem',
  templateUrl: './adminbranditem.component.html',
  styleUrls: ['./adminbranditem.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})

export class AdminbranditemComponent implements OnInit {

  userForm: FormGroup;

  get firstName() {return this.userForm.get("firstName") as FormControl;}
  get lastName() { return this.userForm.get("lastName") as FormControl;}
  get companyName() { return this.userForm.get("companyName") as FormControl;}
  get dba() { return this.userForm.get("dba") as FormControl;}

  //these are from the components as well.

  @Input() urlImageMain: string;
  @Input() fileName: string;
  onlineDescription: string;

  //get bucketName(): string { return this.awsBucket.awsBucket; }
  bucketName:             string;
  awsBucketURL:           string;

  user: IClientTable;
  id: string;
  result: any;

  constructor(public contactService: ContactsService,
        public fb: FormBuilder,
        private router: Router,
        public route: ActivatedRoute,
        private devService: DevService,
        private siteService: SitesService,
        private sanitizer : DomSanitizer,
        private awsBucket: AWSBucketService,
        private clientTableService: ClientTableService,
        private _snackBar: MatSnackBar) {

        this.id = this.route.snapshot.paramMap.get('id');
      }

  childNotifier : Subject<boolean> = new Subject<boolean>();

  async ngOnInit() {

    this.bucketName =   await this.awsBucket.awsBucket();
    this.awsBucketURL = await this.awsBucket.awsBucketURL();
    this.initForm()

  };

  refresh():void {

      this.initForm()

  }

 async update() {

    this.user.firstName =  this.firstName.value
    this.user.lastName =  this.lastName.value
    this.user.companyName = this.companyName.value
    this.user.dba = this.dba.value
    this.user.onlineDescription = this.onlineDescription
    this.user.onlineDescriptionImage = this.urlImageMain

    let promise =  await this.clientTableService.saveClient(this.siteService.getAssignedSite(), this.user).pipe().toPromise()

    if (promise)
    {
      this.notifyEvent("Updated", "Succes")
    } else {
      this.notifyEvent("Failed", "failed")
    }


  };

  initForm(): void {
    //this goes outside of the form, but it's data entry
    //this.onlineDescription = user.onlineDescription

      this.userForm = this.fb.group({
        firstName: '',
        lastName: '',
        companyName: '',
        dba: '',
        id: 0,
      })

      this.clientTableService.getClient(this.siteService.getAssignedSite(), this.id).subscribe(data=> {
        this.user = data
        this.urlImageMain = this.user.onlineDescriptionImage ;
        this.onlineDescription = this.user.onlineDescription ;
        this.userForm.patchValue(this.user)
      })
      console.log(this.urlImageMain)

  };



    received_Image($event) {

      try {
        console.log("received_URLMainImage",  $event)
        let data = $event
        this.urlImageMain = data
        this.update()
      } catch (error) {
        console.log($event)
      }

      console.log("this id ", this.id)
      if (this.id) {
        this.update();
      }

    };

  updateExit() {
    this.update();
    this.goBackToList();
  };
  navBrands() {
    this.goBackToList();
  };

  goBackToList() {
    try {
      this.router.navigate(["adminbrandslist"]);
    } catch (error) {
      console.log(error)
    }
  }


  removeItemFromArray(itemToRemove : string, arrayString : any): any {
    console.log(itemToRemove, arrayString )

    if (!arrayString){
      return ""
    } else {
      let array = arrayString.split(",")
      array.forEach( element => {
        if ( element = itemToRemove ) {
          array.pop()
          return  array.toString()
        }
      });
    }}


    sanitize(html) {
      console.log('sanitize');
      return this.sanitizer.bypassSecurityTrustHtml(html);
    }

    notifyEvent(message: string, action: string) {
      this._snackBar.open(message, action, {
        duration: 2000,
        verticalPosition: 'top'
      });
    }
}
