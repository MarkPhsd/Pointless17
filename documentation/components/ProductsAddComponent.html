import { Component,  Inject,  Input,  OnInit, Optional, } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { DomSanitizer } from '@angular/platform-browser';
import { FormBuilder, FormGroup, Validators} from '@angular/forms';
import { MatSnackBar } from '@angular/material/snack-bar';
import { SitesService } from 'src/app/_services/reporting/sites.service';
import { TaxesService } from 'src/app/_services/menu/taxes.service';
import { AWSBucketService, MenuPricingService,  } from 'src/app/_services';
import { MetrcPackagesService } from 'src/app/_services/metrc/metrc-packages.service';
import { METRCPackagesGet } from 'src/app/_interfaces/metrcs/packages';
import { ISite } from 'src/app/_interfaces/site';
import { tap } from 'rxjs/operators';

@Component({
  selector: 'app-products-add',
  templateUrl: './products-add.component.html',
  styleUrls: ['./products-add.component.scss']
})
export class ProductsAddComponent implements OnInit {

  getPackagesByID
  get f() { return this.packageForm;}
  get bucketName(): string { return this.awsBucket.awsBucket; }
  get awsBucketURL(): string { return this.awsBucket.awsBucketURL; }
  packageForm: FormGroup;
  id: any;
  package: METRCPackagesGet
  site: ISite;

  constructor(
          private router: Router,
          public route: ActivatedRoute,
          public fb: FormBuilder,
          private sanitizer : DomSanitizer,
          private awsBucket: AWSBucketService,
          private _snackBar: MatSnackBar,
          private menuPricingService: MenuPricingService,
          private taxes: TaxesService,
          private siteService: SitesService,
          private metrcPackagesService: MetrcPackagesService)
  {
    this.id = this.route.snapshot.paramMap.get('id');
  }

  async ngOnInit() {
    await this.initForm();
  }

  async initForm() {

    this.packageForm = this.fb.group({
      ProductCategoryType: ['', Validators.required],
      Label: [''],
      ProductionBatchNumber: [''],
      ItemFromFacilityLicenseNumber:  ['', Validators.required],
      ProductName:  ['', Validators.required],
      Invoice:  [''],
      UnitOfMeasureName:  ['', Validators.required],
      Price:  ['', Validators.required],
      id: [0],
      }
    )

      this.site =  this.siteService.getAssignedSite();
      if (this.id) {
        const package$ = this.metrcPackagesService.getPackagesByID(this.id, this.site)
        await package$.subscribe(data => {
          this.package = data
          tap(data => this.packageForm.patchValue(data))
        })
      }
  }


  importStrain() {

  }

}
