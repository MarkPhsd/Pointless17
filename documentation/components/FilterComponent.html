import { Component, OnInit, EventEmitter, Output } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { ReportingService,DashboardService,AuthenticationService } from 'src/app/_services';

export const MY_FORMATS = {
    parse: {
      dateInput: 'LL',
    },
    display: {
      dateInput: 'MM-DD-YYYY',
      monthYearLabel: 'YYYY',
      dateA11yLabel: 'LL',
      monthYearA11yLabel: 'YYYY',
    },
  };

interface IDatePicker {
    reportDateRange: IDaterange
}
interface IDaterange {
    dateFrom: string;
    dateTo: string
}

@Component({
    selector:   'app-widget-filter',
    templateUrl: './filter.component.html',
    styleUrls: [ './filter.component.scss']
  })

export class FilterComponent implements OnInit {


    filterForm: FormGroup;
    calDate: IDatePicker;
    dateFrom: Date;
    dateTo: Date;
    counter: number;
    dateRange: string;

    constructor(
        public fb: FormBuilder,
        private reportingService: ReportingService,
        )
    { }

    @Output() messageOut = new EventEmitter<string>();
    @Output() counterOut = new EventEmitter<number>();

    ngOnInit() {
      this.counter =0;
      this.setFilterDateToday()
      this.emitDatePickerData();  //gets dates from date picker  '

    }

    ngOnChanges() {
      this.onSubmit();
    }

     //sets the default dates
    setFilterDateToday() {
      const today = new Date();
      const month = today.getMonth();
      const year = today.getFullYear();

      this.filterForm =  this.fb.group({
        dateFrom: new Date(year, month, 1),
        dateTo: new Date()
        }
      );

    }


    onSubmit() {
      if (this.filterForm.valid) {
          this.emitDatePickerData()
          return;
      }
    }

    emitDatePickerData() {
        this.counter ++;

        this.dateFrom = this.filterForm.get("dateFrom").value
        this.dateTo = this.filterForm.get("dateTo").value

        //then we want to emit to the Dashboard.
        this.messageOut.emit(this.dateFrom.toLocaleDateString() + ":" + this.dateTo.toLocaleDateString() + ":" + this.counter);

    }

}
