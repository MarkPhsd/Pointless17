import { Component, ViewChild, ChangeDetectorRef, OnInit, Input,
         SimpleChange, OnChanges, SimpleChanges, AfterViewInit,ElementRef } from '@angular/core';
import { MatTableDataSource, MatTable } from '@angular/material/table';
import { MatPaginator, PageEvent} from '@angular/material/paginator';
import { Item,  IUserProfile, }  from 'src/app/_interfaces';
import { Observable, Subject ,fromEvent } from 'rxjs';
import { ActivatedRoute, Router } from '@angular/router';
import { ContactsService, OrdersService } from 'src/app/_services';
import { FormBuilder, FormControl, FormGroup } from '@angular/forms';
import { catchError, delay } from 'rxjs/operators';
import { MatSort } from '@angular/material/sort';
import { IPOSOrder } from 'src/app/_interfaces/transactions/posorder';
import { MatSnackBar } from '@angular/material/snack-bar';
import { SitesService } from 'src/app/_services/reporting/sites.service';
import { debounceTime, distinctUntilChanged, switchMap,filter, tap } from 'rxjs/operators';

@Component({
  selector: 'app-profile-list',
  templateUrl: './profile-list.component.html',
  styleUrls: ['./profile-list.component.scss']
})

export class ProfileListComponent implements OnInit, AfterViewInit {

  @Input() notifier:          Subject<boolean>
  @ViewChild('input', {static: true}) input: ElementRef;
  @ViewChild(MatPaginator) paginator: MatPaginator;
  @ViewChild(MatSort) sort:   MatSort;
  dataSource:                 any;

  item:                       Item; //for routing
  id:                         string;
  public searchForm:                   FormGroup;
  searchPhrase:         Subject<any> = new Subject();

  customerResults$ = this.searchPhrase.pipe(
    debounceTime(250),
    distinctUntilChanged(),
    switchMap(searchPhrase =>
        this.refreshSearch(searchPhrase)
    )
  )

  paginationSize =  50
  currentRow =      1;
  currentPage =     1
  pageNumber =      1
  pageSize =        50
  numberOfPages =   1
  rowCount =        50
  searchPaging:                 boolean;

  search:                       string;
  length:                       number;
  pageSizeOptions:              any;

  product:                      IUserProfile;
  currentCountOfPendingOrders:  any;

  expandedElement:              IUserProfile | null;
  columnsToDisplay = ['id','firstName', 'lastName', 'phone', 'edit', 'checkIn', 'cancelCheckIn', 'openOrder'];

  refreshSearch(searchPhrase):    Observable<IUserProfile[]> {
    const site = this.siteService.getAssignedSite()
    console.log('refreshSearch', searchPhrase)
    this.contactService.getContactBySearch(site, searchPhrase, 1, 50).subscribe(
      data => {
        this.pageSize = 10
        this.length = data.length
        this.pageSizeOptions = [5, 10]
        this.dataSource = new MatTableDataSource(data)
          if (this.dataSource) {
            this.dataSource.paginator = this.paginator;
            this.dataSource.sort = this.sort;
          }
        }
      )
      return   this.contactService.getContactBySearch(site, searchPhrase, 1, 50)
  };


  constructor(private contactService: ContactsService, private cd: ChangeDetectorRef,
              private route: ActivatedRoute,
              private router: Router,
              private fb: FormBuilder,
              private orderService: OrdersService,
              private _snackBar: MatSnackBar,
              private siteService: SitesService,
             ) { }

  // ngOnChanges(changes: { [propKey: string]: SimpleChange }) {

  ngOnInit(): void {
    this.searchForm = this.fb.group({
      searchField: '',
    });
  };

  ngAfterViewInit() {
    fromEvent(this.input.nativeElement,'keyup')
            .pipe(
                filter(Boolean),
                debounceTime(500),
                distinctUntilChanged(),
                tap((event:KeyboardEvent) => {
                  const search  = this.input.nativeElement.value
                  this.refreshSearch(search);
                })
            )
          .subscribe();
  }

  get searchField() {return this.searchForm.get("searchField") as FormControl;}
  get f() { return this.searchForm.controls; }

  editItem(clientID:number) {
    this.router.navigate(["/profileEditor/", {id:clientID}]);
  }

  postNewCheckIn(clientID:number) {

    let order: IPOSOrder;
    const order$ = this.orderService.getNewDefaultCheckIn(this.siteService.getAssignedSite(), clientID)

    order$.subscribe(
      data => {
        // this.findCustomerByPhone()
        this.notifyEvent(`Order Submitted Order # ${data.id}`, "Posted")
      },
      catchError => {
        console.log(catchError)
        this.notifyEvent("Order was not submitted", "Error")
      }
    )
  }

  onSearch(){
    if (this.searchField.value) {
      this.searchPaging = true
      this.pageNumber = 1
      this.currentPage = 1
    } else {
      this.searchPaging = false
    }
    this.refreshSearch(this.searchField.value)
  }

  opeonOrder(orderID: number) {

  }

  async getCountOfPendingOrdersByClient(clientId: number): Promise<any> {

    let count = 0 ;
    const order$ =  this.orderService.getCountOfPendingOrdersByClient(this.siteService.getAssignedSite(), clientId)

    order$.subscribe(
      data => {
        count = data
        return count
    })

    this.currentCountOfPendingOrders = count

  }

  deleteCheckInsOfClient(clientID:number) {

    const numberRemoved = this.orderService.deleteCheckInsOfClient(this.siteService.getAssignedSite(), clientID)

    if (numberRemoved > 0 )  {
      this.notifyEvent("Pending Orders Removed", numberRemoved)
    } else
    {
      this.notifyEvent("Pending Orders Not Removed", "")
    }

  }

  displayFn(search) {
    console.log(search)
    this.selectItem(search)
    this.item = search
    this.search = search
    return search;
  }

  selectItem(search){
    if (search) {
      this.searchPhrase.next(search)
    }
  }

  notifyEvent(message: string, action: string) {
    this._snackBar.open(message, action, {
      duration: 2000,
      verticalPosition: 'top'
    });
  }
}
