    if (name.length) {
      if (name.length >= 5) {
        const realName = name.substring(0, name.length - 4)
        localStorage.setItem(`POSName`, realName)
        if (localStorage.getItem(`POSName`) === realName  ) {
          return true
        } else {
          return false
        }
      }
    } else {
      localStorage.removeItem('POSName');
      return false
    }
