  _isOnBreak(site, user, clock: EmployeeClock): employeeBreak { 
    
    if (!clock) { return null; }
    if ( clock.breaks.length == 0 ) {
      return null;
    }

    clock.breaks.forEach(data => { 
      if (!data.timeEnd) { 
        return data;
      }
    })

    return null;

  }
