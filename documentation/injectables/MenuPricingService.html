import { Injectable, Input } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { AuthenticationService } from '../system/authentication.service';
import { Observable } from 'rxjs';
import { ISite}  from 'src/app/_interfaces';
import { IPriceCategories } from '../../_interfaces/menu/menu-pricing';

@Injectable({
  providedIn: 'root'
})
export class MenuPricingService {


  @Input() currentPage = 1;
  @Input() lastPage:number;
  @Input() pageCount:number;
  @Input() pageSize = 25;
  @Input() pageNumber =  1;

  site: ISite;

  priceCategories$: Observable<IPriceCategories[]>;

  constructor( private http: HttpClient, private auth: AuthenticationService) {
 }

  getPriceCategories(site: ISite): Observable<IPriceCategories[]> {

    const controller = "/pricecategories/"

    const endPoint = `getPriceCategories`

    const parameters = ``

    const url = `${site.url}${controller}${endPoint}${parameters}`

    return  this.http.get<IPriceCategories[]>(url)

  }

  getPriceCategoriesByPage(site: ISite): Observable<IPriceCategories[]> {

    const pageNumber = 1

    const pageSize = 1000

    const controller = "/pricecategories/"

    const endPoint = `getPriceCategories`

    const parameters = `?pageNumber=${pageNumber}&pageSize=${pageSize}`

    const url = `${site.url}${controller}${endPoint}${parameters}`

    return  this.http.get<IPriceCategories[]>(url)

  }

  getPriceCategoriesNoChildrenByPage(site: ISite): Observable<IPriceCategories[]> {

    const pageNumber = 1

    const pageSize = 1000

    const controller = "/pricecategories/"

    const endPoint = `GetPriceCategoriesNoChildrenByPage`

    const parameters = `?pageNumber=${pageNumber}&pageSize=${pageSize}`

    const url = `${site.url}${controller}${endPoint}${parameters}`

    return  this.http.get<IPriceCategories[]>(url)

  }

  getPriceCategory(site: ISite, id: any) : Observable<IPriceCategories> {

    const controller = "/pricecategories/"

    const endPoint = `postPriceCategory`

    const parameters = `?id=${id}`

    const url = `${site.url}${controller}${endPoint}${parameters}`

    return  this.http.get<IPriceCategories>(url)

  }

  putPriceCategory(site: ISite, id: any, model: IPriceCategories) : Observable<IPriceCategories> {

    const controller = "/pricecategories/"

    const endPoint = `putPriceCategory`

    const parameters = `?id=${id}`

    const url = `${site.url}${controller}${endPoint}${parameters}`


    return  this.http.put<IPriceCategories>(url, model)

  }

  postPriceCategory( site: ISite,  model:IPriceCategories) : Observable<IPriceCategories> {

    const controller = "/pricecategories/"

    const endPoint = `postPriceCategory`

    const parameters = ``

    const url = `${site.url}${controller}${endPoint}${parameters}`

    return  this.http.post<IPriceCategories>(url, model)

  }

  savePriceCategory(site: ISite, model: IPriceCategories) : Observable<IPriceCategories> {

    if (model.id !== 0) {

      return this.putPriceCategory(site, model.id, model)

    } else {

      return this.postPriceCategory(site,  model)

    }
  }

}
