  push24HoursToDateArray(date, dateSeries) {
    let i = 0
    const dateString = this.getDateString(date.toDateString())
    for (let i = 0; i < 24; i++) {
      dateSeries.push(
        {date:  `${dateString} ${i}`, value: 0}
      )
      if (i>23) { i = 24}
    }
    return dateSeries;
  }
